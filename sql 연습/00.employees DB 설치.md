```sh
# mysql -p
MariaDB [(none)]> create database employees;
MariaDB [(none)]> create user 'hr'@'192.168.%' identified by 'hr';
MariaDB [(none)]> grant all privileges on employees.* to 'hr'@'192.168.%';
MariaDB [(none)]> flush privileges;
```
